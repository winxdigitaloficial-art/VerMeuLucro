function useSyncedState<T extends Record<string, any>>(key: string, initialValue: T, user: User | null, currentUserEmail: string | null) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    try {
      let item = window.localStorage.getItem(key);
      
      // MIGRATION: Se não tem dados no perfil logado, puxa do perfil offline antigo
      if (!item && key !== STORE) {
        const legacyItem = window.localStorage.getItem(STORE);
        if (legacyItem) {
          item = legacyItem;
          window.localStorage.setItem(key, legacyItem);
        }
      }
      
      return item ? { ...initialValue, ...JSON.parse(item) } : initialValue;
    } catch (error) {
      return initialValue;
    }
  });

  const [syncStatus, setSyncStatus] = useState<'synced' | 'error' | 'syncing'>('syncing');

  // Sync from LocalStorage on mount/key change
  useEffect(() => {
    try {
      let item = window.localStorage.getItem(key);
      if (!item && key !== STORE) {
        const legacyItem = window.localStorage.getItem(STORE);
        if (legacyItem) {
          item = legacyItem;
          window.localStorage.setItem(key, legacyItem);
        }
      }
      if (item) {
        setStoredValue(prev => ({ ...initialValue, ...JSON.parse(item!) }));
      }
    } catch (error) {
      console.error("Local storage error", error);
    }
  }, [key]);

  // Sync from Firestore
  useEffect(() => {
    const docId = currentUserEmail || (user ? user.uid : null);
    if (!docId) return;
    
    setSyncStatus('syncing');
    const docRef = doc(db, "users", docId);
    const unsubscribe = onSnapshot(docRef, (docSnap) => {
      setSyncStatus('synced');
      if (docSnap.exists() && docSnap.data()[key]) {
        const cloudData = docSnap.data()[key];
        const localItem = window.localStorage.getItem(key);
        
        if (localItem) {
          const localData = JSON.parse(localItem);
          const localLen = (localData.lancamentos?.length || 0) + (localData.clientes?.length || 0);
          const cloudLen = (cloudData.lancamentos?.length || 0) + (cloudData.clientes?.length || 0);
          
          if (localLen > cloudLen) {
            // Local data is richer, push it to cloud to avoid data loss
            setDoc(docRef, { [key]: localData }, { merge: true });
            setStoredValue(prev => ({ ...initialValue, ...localData }));
            return;
          }
        }
        
        setStoredValue(prev => ({ ...initialValue, ...cloudData }));
        window.localStorage.setItem(key, JSON.stringify({ ...initialValue, ...cloudData }));
      } else {
        // Nuvem está vazia! Empurra os dados locais para lá automaticamente
        const localItem = window.localStorage.getItem(key);
        if (localItem) {
          const parsed = JSON.parse(localItem);
          if ((parsed.lancamentos && parsed.lancamentos.length > 0) || (parsed.clientes && parsed.clientes.length > 0)) {
            setDoc(docRef, { [key]: parsed }, { merge: true });
          }
        }
      }
    }, (err) => {
      console.warn("Firestore sync disabled/failed. Using local storage only.", err.message);
      setSyncStatus('error');
    });

    return () => unsubscribe();
  }, [user, key, currentUserEmail]);

  const setValue = (value: T | ((val: T) => T)) => {
    try {
      setStoredValue(prev => {
        const valueToStore = value instanceof Function ? value(prev) : value;
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
        
        // Sync to Firestore
        const docId = currentUserEmail || (user ? user.uid : null);
        if (docId) {
          setSyncStatus('syncing');
          const docRef = doc(db, "users", docId);
          // Strip undefined values which crash Firestore
          const safeValue = JSON.parse(JSON.stringify(valueToStore));
          setDoc(docRef, { [key]: safeValue }, { merge: true }).then(() => {
            setSyncStatus('synced');
          }).catch(err => {
            console.warn("Failed to sync to Firestore", err.message);
            setSyncStatus('error');
          });
        }
        return valueToStore;
      });
    } catch (error) {
      console.log(error);
    }
  };

  return [storedValue, setValue, syncStatus] as const;
}

function fmtBRL(v: number) {
  return 'R$ ' + Math.abs(v).toLocaleString('pt-BR', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
}

function fmtBRLShort(v: number) {
  if (v >= 1000) return 'R$ ' + (v / 1000).toFixed(1).replace('.', ',') + 'k';
  return 'R$ ' + Math.round(v);
}

function getMesNome(ym: string) {
  const [y, m] = ym.split('-');
  const meses = ['Janeiro', 'Fevereiro', 'Março', 'Abril', 'Maio', 'Junho', 'Julho', 'Agosto', 'Setembro', 'Outubro', 'Novembro', 'Dezembro'];
  return meses[parseInt(m) - 1] + ' ' + y;
}

function fmtData(d: string) {
  if (!d) return '';
  const [, m, dia] = d.split('-');
  return dia + '/' + m;
}

                  <div className="text-muted ml-2">
                    <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="m9 18 6-6-6-6"/></svg>
                  </div>
                </button>
              ))
            )}
          </div>
        </div>
      </div>
    );
  };

  const [suporteMsg, setSuporteMsg] = useState('');
  const [chatMessages, setChatMessages] = useState<{role: 'user'|'ai', text: string}[]>([
    {role: 'ai', text: 'Olá! Sou seu assistente virtual. Como posso ajudar você a usar o Vermeulucro hoje?'}
  ]);

  const handleSendSuporte = (e: React.FormEvent) => {
    e.preventDefault();
    if (!suporteMsg.trim()) return;
    
    setChatMessages(prev => [...prev, {role: 'user', text: suporteMsg}]);
    setSuporteMsg('');
    
    // Placeholder para a resposta da IA (a ser integrada depois)
    setTimeout(() => {
      setChatMessages(prev => [...prev, {role: 'ai', text: 'Entendi sua dúvida! Nossa integração com IA está sendo configurada e em breve poderei te responder com precisão. Por enquanto, se precisar de ajuda urgente, chame no WhatsApp do suporte.'}]);
    }, 1000);
  };

  const renderSuporte = () => (
    <div id="tab-suporte" className="flex flex-col h-[calc(100vh-180px)] max-h-[600px]">
      <div className="card flex-1 flex flex-col mb-0 overflow-hidden p-0 relative">
        <div className="bg-card2 p-4 border-b border-border flex items-center gap-3">
          <div className="w-10 h-10 rounded-full bg-green-900/30 text-green-500 flex items-center justify-center text-xl">
            🤖
          </div>
          <div>
            <div className="font-bold text-white">Suporte Inteligente</div>
            <div className="text-xs text-muted">Tire dúvidas sobre o sistema</div>
          </div>
        </div>
        
        <div className="flex-1 overflow-y-auto p-4 flex flex-col gap-4">
          {chatMessages.map((msg, idx) => (
            <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
              <div className={`max-w-[85%] p-3 rounded-2xl text-sm leading-relaxed ${msg.role === 'user' ? 'bg-green-600 text-white rounded-tr-sm' : 'bg-card2 text-white border border-border rounded-tl-sm'}`}>
                {msg.text}
              </div>
            </div>
          ))}
        </div>
        
        <div className="p-4 bg-surface border-t border-border mt-auto">
          <form onSubmit={handleSendSuporte} className="flex gap-2 relative">
            <input 
              type="text" 
              className="form-input flex-1 pr-12 bg-card border-border focus:border-green-500" 
              placeholder="Digite sua dúvida aqui..." 
              value={suporteMsg}
              onChange={e => setSuporteMsg(e.target.value)}
            />
            <button type="submit" className="absolute right-2 top-1/2 -translate-y-1/2 p-2 text-green-500 hover:text-green-400 transition-colors bg-transparent border-none">
              <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="m22 2-7 20-4-9-9-4Z"/><path d="M22 2 11 13"/></svg>
            </button>
          </form>
        </div>
      </div>
    </div>
  );

  const renderLogin = () => (
    <div className="flex min-h-screen items-center justify-center bg-[#060D09] p-4">
      <div className="card w-full max-w-md">
        <div className="text-center mb-6">
          <div className="logo" style={{ fontSize: '1.8rem', padding: 0, height: 'auto' }}>Ver<span>Meu</span>Lucro</div>
          <div className="text-muted mt-2">
            {esqueciSenha ? 'Recuperar Acesso' : isRegistering ? 'Crie sua conta' : 'Faça login para continuar'}
          </div>
        </div>
        <form onSubmit={handleAuth}>
          <div className="form-group">
            <label className="form-label">E-mail</label>
            <input type="email" className="form-input" value={loginEmail} onChange={e => setLoginEmail(e.target.value)} required />
          </div>
          
          {!esqueciSenha && (
            <div className={`form-group ${isRegistering ? '' : 'mb-6'}`}>
              <label className="form-label">Senha</label>
              <div className="relative">
                <input 
                  type={showPassword ? "text" : "password"} 
                  className="form-input" 
                  value={loginSenha} 
                  onChange={e => setLoginSenha(e.target.value)} 
                  required 
                />
              </div>
              <div className="flex items-center gap-2 mt-2">
                <input 
                  type="checkbox" 
                  id="showPassword" 
                  checked={showPassword} 
                  onChange={() => setShowPassword(!showPassword)} 
                  className="w-4 h-4 accent-green-500 rounded bg-card border-border"
                />
                <label htmlFor="showPassword" className="text-sm text-muted cursor-pointer select-none hover:text-white transition-colors">
                  Mostrar senha
                </label>
              </div>
            </div>
          )}

          {isRegistering && !esqueciSenha && (
            <div className="form-group mb-6">
              <label className="form-label">Dica de Senha (opcional)</label>
              <input type="text" className="form-input" value={loginDica} onChange={e => setLoginDica(e.target.value)} placeholder="Ex: Nome do meu cachorro" />
            </div>
          )}

          {loginError && (
            <div className={`text-sm text-center mb-4 p-3 rounded-lg ${loginError.includes('Dica') ? 'bg-green-900/30 text-green-400' : 'bg-red-900/30 text-red-400'}`}>
              {loginError}
            </div>
          )}

          <button type="submit" className="btn btn-green mb-4">
            {esqueciSenha ? 'Ver Dica de Senha' : isRegistering ? 'Cadastrar' : 'Entrar com E-mail'}
          </button>

          {!esqueciSenha && (
            <>
              <div className="relative mb-4 flex items-center py-2">
                <div className="flex-grow border-t border-border"></div>
                <span className="flex-shrink-0 mx-4 text-muted text-xs uppercase tracking-wider">Ou</span>
                <div className="flex-grow border-t border-border"></div>
              </div>
              
              <button type="button" onClick={handleGoogleLogin} className="btn bg-white text-black hover:bg-gray-200 mb-6 flex items-center justify-center gap-2">
                <svg width="18" height="18" viewBox="0 0 24 24" fill="none" xmlns="http://www.w3.org/2000/svg">
                  <path d="M22.56 12.25c0-.78-.07-1.53-.2-2.25H12v4.26h5.92c-.26 1.37-1.04 2.53-2.21 3.31v2.77h3.57c2.08-1.92 3.28-4.74 3.28-8.09z" fill="#4285F4"/>
                  <path d="M12 23c2.97 0 5.46-.98 7.28-2.66l-3.57-2.77c-.98.66-2.23 1.06-3.71 1.06-2.86 0-5.29-1.93-6.16-4.53H2.18v2.84C3.99 20.53 7.7 23 12 23z" fill="#34A853"/>
                  <path d="M5.84 14.09c-.22-.66-.35-1.36-.35-2.09s.13-1.43.35-2.09V7.07H2.18C1.43 8.55 1 10.22 1 12s.43 3.45 1.18 4.93l2.85-2.22.81-.62z" fill="#FBBC05"/>
                  <path d="M12 5.38c1.62 0 3.06.56 4.21 1.64l3.15-3.15C17.45 2.09 14.97 1 12 1 7.7 1 3.99 3.47 2.18 7.07l3.66 2.84c.87-2.6 3.3-4.53 6.16-4.53z" fill="#EA4335"/>
                </svg>
                Entrar com Google
              </button>
            </>
          )}
          
          <div className="flex flex-col gap-3 text-center">
            {!esqueciSenha && !isRegistering && (
              <button type="button" onClick={() => { setEsqueciSenha(true); setLoginError(''); }} className="text-muted hover:text-white text-sm transition-colors">
                Esqueci minha senha
              </button>
            )}
            <button type="button" onClick={() => { setIsRegistering(!isRegistering); setEsqueciSenha(false); setLoginError(''); }} className="text-muted hover:text-white text-sm transition-colors">
              {isRegistering || esqueciSenha ? 'Voltar para o Login' : 'Não tem conta? Cadastre-se'}
            </button>
          </div>
        </form>
      </div>
    </div>
  );

  if (!currentUser) return renderLogin();

return (
    <div className="app">
      <div className="topbar">
        <div className="logo flex items-center gap-2">
          Ver<span>Meu</span>Lucro
          {syncStatus === 'synced' && <span className="text-green-500 text-xs flex items-center gap-1" title="Nuvem sincronizada"><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M17.5 19H9a7 7 0 1 1 6.71-9h1.79a4.5 4.5 0 1 1 0 9Z"/></svg></span>}
          {syncStatus === 'error' && <span className="text-red-500 text-xs flex items-center gap-1" title="Erro na nuvem (Verifique o Firestore)"><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M17.5 19H9a7 7 0 1 1 6.71-9h1.79a4.5 4.5 0 1 1 0 9Z"/><line x1="12" x2="12" y1="8" y2="12"/><line x1="12" x2="12.01" y1="16" y2="16"/></svg></span>}
          {syncStatus === 'syncing' && <span className="text-amber-500 text-xs flex items-center gap-1" title="Sincronizando..."><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className="animate-spin"><path d="M21 12a9 9 0 1 1-6.219-8.56"/></svg></span>}
        </div>
        <div className="flex items-center gap-4 flex-wrap justify-end">
          <div className="text-xs text-muted bg-card2 px-2 py-1 rounded border border-border flex items-center gap-1" title="E-mail logado">
            👤 {currentUser}
          </div>
          <div className="month-nav">
            <button className="month-btn" onClick={() => mudarMes(-1)}>&#8249;</button>
            <span className="month-label">{getMesNome(mesAtual)}</span>
            <button className="month-btn" onClick={() => mudarMes(1)}>&#8250;</button>
          </div>
          <button onClick={handleLogout} className="text-muted hover:text-red-500 text-sm font-medium transition-colors">Sair</button>
        </div>
      </div>

      <div className="content">
        <div className="tabs">
          <button className={`tab ${activeTab === 'dashboard' ? 'active' : ''}`} onClick={() => setActiveTab('dashboard')}>Dashboard</button>
          <button className={`tab ${activeTab === 'lancar' ? 'active' : ''}`} onClick={() => setActiveTab('lancar')}>Lançar</button>
          <button className={`tab ${activeTab === 'areceber' ? 'active' : ''}`} onClick={() => setActiveTab('areceber')}>A Receber</button>
          <button className={`tab ${activeTab === 'clientes' ? 'active' : ''}`} onClick={() => setActiveTab('clientes')}>Clientes</button>
          <button className={`tab ${activeTab === 'horas' ? 'active' : ''}`} onClick={() => setActiveTab('horas')}>Horas</button>
          <button className={`tab ${activeTab === 'relatorio' ? 'active' : ''}`} onClick={() => setActiveTab('relatorio')}>Relatório</button>
          <button className={`tab ${activeTab === 'suporte' ? 'active' : ''}`} onClick={() => setActiveTab('suporte')}>Suporte</button>
        </div>

        {activeTab === 'dashboard' && renderDashboard()}
        {activeTab === 'lancar' && renderLancar()}
        {activeTab === 'areceber' && renderAReceber()}
        {activeTab === 'clientes' && renderClientes()}
        {activeTab === 'horas' && renderHoras()}
        {activeTab === 'relatorio' && renderRelatorio()}
        {activeTab === 'suporte' && renderSuporte()}
      </div>
    </div>
  );
}
